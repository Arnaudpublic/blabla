Exercice 5

float r = 1;
float circle_size = 40;

void setup()
{ size(400, 400);
background(0, 0, 0);
smooth();
noStroke();}

void draw()
{ fill(255);
rotate(r);
float circle_size = random(10, 10);
ellipse(10 + r, 10, circle_size, circle_size);
r = r + 1;
println(r); }
